git init - инициализация
git add - добавление файлов в гит (git add *) - добавить всё
git status - покажет статус и скажет что делать
git config --global user.name "Name" - подпись для идентификации себя
git config --global user.email "e@m.l" -  - подпись для идентификации себя
git commit -m "new:create somethink" - описание своих изменений
git config --list - проверка всех данных пользователя

https://git-scm.com/book/ru - книга по гит

git log - просмотр изменений
git checkout 'SHA - вернуться к какомуто коммиту(sha)
git checkout master - вернет к мастеру
git remote add origin https://github.com/user/project.git - добавляем удаленный репозиторий
git fetch origin - скопируем состояние удаленного репозитория и синхронизация с удаленным
git rebase origin/master - объединение локального и удаленного репозитория (не очень хорошая команда)
git push origin master - запись в удаленный репозиторий
git clone - копирование репозитория
git merge something - сливание веток проекта(какую ветку будем вливать в текущую)
